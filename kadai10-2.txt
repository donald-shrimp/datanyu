void insertion(int x[], int num) 
{
  int i,j,tmp;
  for (i=1;i<=num-1;i++) {
    tmp=x[i];
    j=i;
    while ((x[j-1]>tmp) && (j>0)) {
      x[j]=x[j-1];
      j--;
    }
    x[j]=tmp;
  }                                                    
}
